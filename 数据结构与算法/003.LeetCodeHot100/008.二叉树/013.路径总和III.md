# 路径总和 III - 回溯

```go
/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */

var (
	ans int
	theTarget int
	sums []int
)

func pathSum(root *TreeNode, targetSum int) int {
	ans = 0
	theTarget = targetSum
	sums = []int{}
	process(root)

	return ans >> 1
}

func process(root *TreeNode) {
	for _, sum := range sums {
		if sum == theTarget {
			ans++
		}
	}
	if root == nil {
		return
	}
	for i := range sums {
		sums[i] += root.Val
	}
	sums = append(sums, root.Val)
	process(root.Left)
	process(root.Right)
	sums = sums[:len(sums) - 1]
	for i := range sums {
		sums[i] -= root.Val
	}
}
```

# 路径总和 III - 深度优先遍历

```go
/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func pathSum(root *TreeNode, targetSum int) int {
	if root == nil {
		return 0
	}

	ans := rootSum(root, targetSum)
	ans += pathSum(root.Left, targetSum)
	ans += pathSum(root.Right, targetSum)

	return ans
}

// 以节点 root 为起点向下且满足路径总和为 targetSum 的路径数目
func rootSum(root *TreeNode, targetSum int) (ans int) {
	if root == nil {
		return
	}

	sub := targetSum - root.Val
	if sub == 0 {
		ans++
	}
	ans += rootSum(root.Left, sub)
	ans += rootSum(root.Right, sub)
	return
}
```
